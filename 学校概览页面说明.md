**参考代码结构：**
```javascript
let schoolChartsInitialized = false;

function switchRecordTab(tab) {
    const tabs = ['school', 'grade', 'class', 'student', 'records', 'scores', 'rank', 'plan'];
    tabs.forEach(t => {
        const btn = document.getElementById('rtab-' + t);
        const view = document.getElementById('view-' + t);
        if (!btn || !view) return;
        if (t === tab) {
            btn.classList.remove('border-transparent', 'text-gray-500');
            btn.classList.add('border-blue-600', 'text-blue-600');
            view.classList.remove('hidden');
            
            // 切换到学校概览时初始化图表
            if (t === 'school' && !schoolChartsInitialized) {
                setTimeout(() => {
                    initSchoolOverviewCharts();
                    schoolChartsInitialized = true;
                }, 100);
            }
        } else {
            btn.classList.add('border-transparent', 'text-gray-500');
            btn.classList.remove('border-blue-600', 'text-blue-600');
            view.classList.add('hidden');
        }
    });
    lucide.createIcons();
}

function initSchoolOverviewCharts() {
    // 1. 测试人数 - 环形图
    const chartTestCount = echarts.init(document.getElementById('chart-test-count'));
    chartTestCount.setOption({
        series: [{
            type: 'pie',
            radius: ['50%', '70%'],
            label: {
                show: true,
                position: 'center',
                formatter: '43%',
                fontSize: 20
            },
            data: [
                { value: 533, name: '已测试', itemStyle: { color: '#3b82f6' } },
                { value: 700, name: '未测试', itemStyle: { color: '#e5e7eb' } }
            ]
        }]
    });
    
    // 2-11. 其他图表...
    
    // 响应式
    window.addEventListener('resize', () => {
        chartTestCount.resize();
        // ...其他图表的resize
    });
}
```
